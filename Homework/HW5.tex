	$$\begin{align}
		y''+y'-2y&=x^2\\
		-2y&=x^2-y'-y''\\
		y&=-\frac{2}{3}x^2+\frac{1}{2}y'+\frac{1}{2}y''
	\end{align}\qquad

		2 &= 4(0)^3 + 3a(0)^2 + 2b(0) + c &\qquad 1 &= 0^4 + a(0)^3 + b(0)^2 + c(0) + d \\
